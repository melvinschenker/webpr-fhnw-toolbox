Lambda Essentials

Fundamentale Higher-Order Functions der funktionalen Programmierung: Curry, Pipe und Compose.

Curry

// Curry: f(a,b,c) zu f(a)(b)(c)
const curry = f => (...xs) =>
  xs.length >= f.length ? f(...xs) : (...ys) => curry(f)(...xs, ...ys);

const add = (a, b) => a + b;
const add3 = curry(add)(3);
console.log(add3(5));  // 8

Pipe

// Pipe: L채sst Funktionen von links nach rechts ausf체hren.
const pipe = (...fs) => x => fs.reduce((v, f) => f(v), x);

const times = (a, b) => a * b;
const add3 = curry(add)(3);
const times2 = curry(times)(2);

console.log(pipe(add3, times2)(5));  // (5 + 3) * 2 = 16

Compose

// Compose: L채sst Funktionen von rechts nach links ausf체hren.
const compose = (...fs) => x => fs.reduceRight((v, f) => f(v), x);

const f = x => x + 1;
const g = x => x * 2;
const h = compose(f, g);  // f(g(x))
console.log(h(5));  // f(g(5)) = f(10) = 11

Beispiel: Username-Validierung

const trim = str => str.trim();
const toLowerCase = str => str.toLowerCase();
const removeSpaces = str => str.replace(/\s+/g, '');
const minLength = curry((min, str) => str.length >= min ? str : null);
const maxLength = curry((max, str) => str.length <= max ? str : null);

// Username-Validierung

  trim,
  toLowerCase,
  removeSpaces,
